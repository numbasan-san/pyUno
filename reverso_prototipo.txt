
players = ['Bot_0', 'Bot_1', 'Bot_2', 'Bot_3', 'Bot_4', 'Bot_5', 'Bot_6']
game = True

print(players)

def find_index_player(player):
  index_list = []
  index = players.index(player)
  print(f'El jugador en turno, {player}, está en el puesto {index} de la lista')
  
  if index == (len(players) - 1):
    print('final alcanzado')
    for i in range(index):
      index_list.append(i)
    index_list.append(index)
    index_list = index_list[::-1]
      
  elif index > 0 and index < len(players):
    print('No es mucho, pero es trabajo honesto')
    
    print(f'A ver... {list(set(players[index]).difference(players))}')
    
    for i in range(0, (index)):
      index_list.append(i)
    index_list = index_list[::-1]
    
    for i in range(index, (len(players))):
      index_list.append(i)
    index_list = index_list[::-1]

  elif index == 0:
    print('inicio alcanzado')
    for i in range(1, len(players)):
      index_list.append(i)
    index_list.append(index)
    index_list = index_list[::-1]

  return index_list

def play(player_in_turn):
  print(find_index_player(player_in_turn))
  print()

def pregunta(text, opts):
    result = ''
    while result not in opts:
        result = input(text)
    return result
    
while game:
      
  for player in players:
      play(player)
      # print(player)

  print(players)
